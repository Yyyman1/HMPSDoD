		#	this = c:DAI
		#}

		any_primary_culture = {

				has_discrimination_trait = mongol_language
				has_discrimination_trait = sinosphere
			}
			
		}

		#	}
		#any_scope_state = {

		#					OR = {
		#						is_homeland = cu:han

		#				
		#			}
	}

    possible = {
		
		#has_law = law_type:law_militarized_police
		has_law = law_type:law_appointed_bureaucrats
		#country_has_voting_franchise = yes
		NOT = {
			has_law = law_type:law_peonage
			OR = {
				has_law = law_type:law_autocracy
				has_law = law_type:law_oligarchy
				has_law = law_type:law_technocracy

				type = je_china_reunification_outsider
			}
	
		
    }

		base = 100
	}
}

dvg_annex_siberia = {
	is_shown = {
	
		NOT = { is_country_type = decentralized }
		exists = c:SIB



		any_primary_culture = {
			has_discrimination_trait = north_asian_heritage
		}
		any_neighbouring_state = {
			owner = {is_country_type = decentralized}
		}
		has_interest_marker_in_region = sr:region_west_siberia
		
	}

    possible = {

		OR = {
			#is_player = yes
			year >= 1900
			is_country_type = recognized
			has_technology_researched = pan-nationalism
		}
		has_technology_researched = colonization
		NOT = { has_law = law_type:law_no_colonial_affairs }
		army_size >= 20
		
	
	}

    when_taken = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					is_country_type = decentralized
					any_primary_culture = {
						has_discrimination_trait = north_asian_heritage
						
					}
					any_scope_state = {
						region = sr:region_west_siberia
					}

				}			
				root = { annex = prev }
			}
		}
		if = {
			limit = {
				exists = c:SBR
			}
			create_diplomatic_play = {
				type = dp_annex_war
				target_country = c:SBR

				add_war_goal = {
					holder = c:SBR
					target_country = ROOT
					type = war_reparations
				}
			}
		}
		
		create_incident = {
			country = ROOT
			value = 5
		}
    }

	ai_chance = {
		base = 100
	}
}
