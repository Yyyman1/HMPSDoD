je_chinese_industry = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	group = je_group_historical_content
	
	complete = {
		custom_tooltip = {
			text = je_china_economic_success_tt
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}
	
	on_complete = {
		remove_modifier = chinese_economic_stagnation
	}

	on_monthly_pulse = {
		effect = {
			### Scaling by Railroads
			
			set_variable = {
				name = railroad_effect
				value = global_var:chinese_rail_expansion
			}
			
			change_variable = {
				name = railroad_effect
				divide = -100
			}
			
			change_variable = {
				name = railroad_effect
				add = 1
			}
			
			remove_modifier = chinese_economic_stagnation
			
			add_modifier = {
				name = chinese_economic_stagnation
				multiplier = var:railroad_effect
			}
		}
	}

	current_value = {
		value = global_var:chinese_rail_expansion
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	can_revolution_inherit = yes

	weight = 5000
	should_be_pinned_by_default = yes
}
